\documentclass{article}

% Import custom style.
\usepackage{../.preamble/tikz_diagrams_template}

\colortheme{blue}

% ---------
% Document.

\begin{document}

    % Sets viewing orientation (declination/rotation) of 3D coordinate system.
    \tdplotsetmaincoords{70}{110}

    % -----------------
    % TikZ environment.
    % -----------------

    \begin{tikzpicture}[scale=0.7,tdplot_main_coords]
        
        % sets dimensions
        \pgfmathsetmacro{\a}{4}
        \pgfmathsetmacro{\b}{5}
        \pgfmathsetmacro{\c}{3}
        \pgfmathsetmacro{\axislength}{5}
        
        % draws rectangular prism
        \draw[style=dashed,color=gray](0,0,0)--(\a,0,0);
        \draw[style=dashed,color=gray](0,0,0)--(0,\b,0);
        \draw[style=dashed,color=gray](0,0,0)--(0,0,\c);
        \draw(\a,0,0)--(\a,\b,0);
        \draw(0,\b,0)--(\a,\b,0);
        \draw(0,0,\c)--(\a,0,\c);
        \draw(\a,0,0)--(\a,0,\c);
        \draw(0,\b,0)--(0,\b,\c);
        \draw(0,0,\c)--(0,\b,\c);
        \draw(0,\b,\c)--(\a,\b,\c);
        \draw(\a,0,\c)--(\a,\b,\c);
        \draw(\a,\b,0)--(\a,\b,\c);
        
        % labels faces
        \draw[color=gray](\a/2,0,\c/2)node{$1$};
        \draw(\a/2,\b,\c/2+0.3)node{$2$};
        \draw[color=gray](0,\b/2+0.3,\c/2+0.5)node{$3$};
        \draw(\a,\b/2,\c/2+0.5)node{$4$};
        \draw[color=gray](\a/2,\b/2,0)node{$5$};
        \draw(\a/2,\b/2-0.3,\c)node{$6$};
        
        % draws body axes
        \draw[color=red](\a/2,\b/2,\c/2)--(\a/2+\axislength,\b/2,\c/2)node[anchor=west,xshift=1]{$x_{i}$};
        \draw[color=green](\a/2,\b/2,\c/2)--(\a/2,\b/2+\axislength,\c/2)node[anchor=south,color=green!75!black]{$y_{i}$};
        \draw[color=blue](\a/2,\b/2,\c/2)--(\a/2,\b/2,\c/2+\axislength)node[anchor=west,xshift=-1]{$z_{i}$};
        
        % labels dimensions
        \draw(\a/2,\b+0.4,0)node{$a_{i}$};
        \draw(\a,\b/2,-0.4)node{$b_{i}$};
        \draw(0,\b+0.4,\c/2)node{$c_{i}$};
        
    \end{tikzpicture}
    
\end{document}
