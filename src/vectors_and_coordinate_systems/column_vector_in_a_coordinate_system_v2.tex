    \begin{tikzpicture}[tdplot_main_coords,scale=1]

        \node[xshift=1,yshift=-6]at(0,0){$O$};
        
        % axes
        \draw(0,0,0)--(5,0,0)node[pos=1.07]{$x_{1}$};

        \draw(0,0,0)--(0,0,5)node[pos=1.035]{$x_{3}$};

        \draw[-mylatex',thick](0,0,0)--(0,1.5,0)node[below,xshift=-15,yshift=3]{$\mathbf{\hat{x}}_{2}$};

    \end{tikzpicture}
